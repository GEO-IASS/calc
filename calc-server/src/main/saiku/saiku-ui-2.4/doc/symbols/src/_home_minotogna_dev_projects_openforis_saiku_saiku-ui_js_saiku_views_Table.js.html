 21 /**
 22  * Class which handles table rendering of resultsets
 23  */
 24 var Table = Backbone.View.extend({
 25     tagName: "table",
 26 
 27     events: {
 28         'click .cancel': 'cancel',
 29         'click th.row' : 'clicked_cell',
 30         'click th.col' : 'clicked_cell'
 31     },
 32 
 33     initialize: function(args) {
 34         this.workspace = args.workspace;
 35         
 36         // Bind table rendering to query result event
 37         _.bindAll(this, "render", "process_data", "cancelled", "cancel");
 38         this.workspace.bind('query:result', this.render);
 39     },
 40     
 41     clicked_cell: function(event) {
 42         var self = this;
 43         
 44         if (this.workspace.query.get('type') != 'QM' || Settings.MODE == "view") {
 45             return false;
 46         }
 47 
 48         $target = ($(event.target).hasClass('row') || $(event.target).hasClass('col') ) ?
 49             $(event.target).find('div') : $(event.target);
 50         
 51     $body = $(document);
 52     $body.off('.contextMenu .contextMenuAutoHide');
 53     $('.context-menu-list').remove();
 54     $.contextMenu('destroy');
 55     $.contextMenu({
 56         appendTo: $target,
 57         selector: '.row, .col', 
 58         ignoreRightClick: true,
 59          build: function($trigger, e) {
 60             $target = $(e.currentTarget).find('div');
 61             var axis = $(e.currentTarget).hasClass('rows') ? "ROWS" : "COLUMNS"
 62             var pos = $target.attr('rel').split(':');
 63             var row = parseInt(pos[0])
 64             var col = parseInt(pos[1])
 65             var cell = self.workspace.query.result.lastresult().cellset[row][col];
 66             var query = self.workspace.query;
 67             var schema = query.get('schema');
 68             var cube = query.get('connection') + "/" + 
 69                 query.get('catalog') + "/"
 70                 + ((schema == "" || schema == null) ? "null" : schema) 
 71                 + "/" + query.get('cube');
 72 
 73             var d = cell.properties.dimension;
 74             var h = cell.properties.hierarchy;
 75             var l = cell.properties.level;
 76 
 77             var keep_payload = JSON.stringify(
 78                 {
 79                     "hierarchy"     : "[" + h + "]",
 80                     "uniquename"    : l,
 81                     "type"          : "level",
 82                     "action"        : "delete"
 83                 }) 
 84             + "," +JSON.stringify(
 85                 {
 86                     "hierarchy"     : "[" + h + "]",
 87                     "uniquename"    : cell.properties.uniquename,
 88                     "type"          : "member",
 89                     "action"        : "add"
 90                 }       
 91             );
 92 
 93             var levels = [];
 94             var items = {};
 95             var dimensions = Saiku.session.sessionworkspace.dimensions[cube].get('data');
 96             var dimsel = {};
 97             var used_levels = [];
 98 
 99             self.workspace.query.action.get("/axis/" + axis + "/dimension/" + d, { 
100                         success: function(response, model) {
101                             dimsel = model;
102                         },
103                         async: false
104             });
105 
106             _.each(dimsel.selections, function(selection) {
107                 if(_.indexOf(used_levels, selection.levelUniqueName) == -1)
108                     used_levels.push(selection.levelUniqueName);
109 
110             });
111 
112             _.each(dimensions, function(dimension) {
113                 if (dimension.name == d) {
114                     _.each(dimension.hierarchies, function(hierarchy) {
115                         if (hierarchy.name == h) {
116                             _.each(hierarchy.levels, function(level) {
117                                 items[level.name] = {
118                                     name: level.name,
119                                     payload: JSON.stringify({
120                                         "hierarchy"     : "[" + h + "]",
121                                         uniquename    : level.uniqueName,
122                                         type          : "level",
123                                         action        : "add"
124                                     })
125                                 };
126                                 if(_.indexOf(used_levels, level.uniqueName) > -1) {
127                                     items[level.name].disabled = true;
128                                     items["remove-" + level.name] = {
129                                         name: level.name,
130                                         payload: JSON.stringify({
131                                             "hierarchy"     : "[" + h + "]",
132                                             uniquename    : level.uniqueName,
133                                             type          : "level",
134                                             action        : "delete"
135                                         })
136                                     };
137                                     
138                                 }
139                                 items["keep-" + level.name] = items[level.name];
140                                 items["include-" + level.name] = JSON.parse(JSON.stringify(items[level.name]));
141                                 items["keep-" + level.name].payload = keep_payload + "," + items[level.name].payload;
142                             });
143                         }
144                     });
145                 }
146             });
147             items["keeponly"] = { payload: keep_payload }
148             
149 
150             
151             var lvlitems = function(prefix) {
152                 var ritems = {};
153                 for (key in items) {
154                     if (prefix != null && prefix.length < key.length && key.substr(0, prefix.length) == prefix) {
155                             ritems[key] = items[key];
156                     }
157                 }
158                 return ritems;
159             }
160 
161             var member = $target.html();
162 
163             var citems = {
164                     "name" : {name: "<b>" + member + "</b>", disabled: true },
165                     "sep1": "---------",
166                     "keeponly": {name: "Keep Only", payload: keep_payload }
167             };
168             if (d != "Measures") {
169                 citems["fold1key"] = {
170                         name: "Include Level",
171                         items: lvlitems("include-")
172                     };
173                 citems["fold2key"] = {
174                         name: "Keep and Include Level",
175                         items: lvlitems("keep-")
176                     };
177                 citems["fold3key"] = {
178                         name: "Remove Level",
179                         items: lvlitems("remove-")
180                     };
181             }
182             return {
183                 callback: function(key, options) {
184                     self.workspace.query.action.put('/axis/' + axis + '/dimension/' + d, { 
185                         success: function() {
186                             self.workspace.query.clear();
187                             self.workspace.query.fetch({ success: function() {
188                                 
189                                 $(self.workspace.el).find('.fields_list_body ul').empty();
190                                 $(self.workspace.dimension_list.el).find('.parent_dimension a.folder_collapsed').removeAttr('style');
191                                 
192                                 $(self.workspace.dimension_list.el).find('.parent_dimension ul li')
193                                     .draggable('enable')
194                                     .css({ fontWeight: 'normal' });
195 
196                                 $(self.workspace.measure_list.el).find('a.measure').parent()
197                                     .draggable('enable')
198                                     .css({ fontWeight: 'normal' });
199 
200                                 self.workspace.populate_selections(self.workspace.measure_list.el);
201                                 $(self.workspace.el).find('.fields_list_body ul li')
202                                     .removeClass('ui-draggable-disabled ui-state-disabled')
203                                     .css({ fontWeight: 'normal' });
204 
205                              }});
206 
207                         },
208                         data: {
209                             selections: "[" + items[key].payload + "]"
210                         }
211                     });
212                     
213                 },
214                 items: citems
215             } 
216         }
217     });
218     $target.contextMenu();
219 
220 
221     },

224     render: function(args, block) {
225 
226         $(this.workspace.el).find(".workspace_results_info").empty();
227 
228         if (args.data != null && args.data.error != null) {
229             return this.error(args);
230         }
231 
232         
233         // Check to see if there is data
234         if (args.data.cellset && args.data.cellset.length === 0) {
235             return this.no_results(args);
236         }
237         
238         // Clear the contents of the table
239         var runtime = args.data.runtime != null ? (args.data.runtime / 1000).toFixed(2) : "";
240         $(this.workspace.el).find(".workspace_results_info")
241             .html('<b>Rows:</b> ' + args.data.height + " <b>Columns:</b> " + args.data.width + " <b>Duration:</b> " + runtime + "s");
242         $(this.el).html('<tr><td>Rendering ' + args.data.width + ' columns and ' + args.data.height + ' rows...</td></tr>');
243 
244         // Render the table without blocking the UI thread
245         if (block === true) {
246             this.process_data(args.data.cellset);
247         } else {
248             _.delay(this.process_data, 0, args.data.cellset);
249         }
250 
251     },
252 
253     process_data: function(data) {
254         var contents = "";
255         var table = data ? data : [];
256         var colSpan;
257         var colValue;
258         var isHeaderLowestLvl;
259         var isBody = false;
260         var firstColumn;
261         var isLastColumn, isLastRow;
262         var nextHeader;
263         var processedRowHeader = false;
264         var lowestRowLvl = 0;
265         var rowGroups = [];
266 
267         for (var row = 0; row < table.length; row++) {
268             colSpan = 1;
269             colValue = "";
270             isHeaderLowestLvl = false;
271             isLastColumn = false;
272             isLastRow = false;
273             headerStarted = false;
274 
275             contents += "<tr>";
276 
277             for (var col = 0; col < table[row].length; col++) {
278                 var header = data[row][col];
279 
280                 // If the cell is a column header and is null (top left of table)
281                 if (header.type === "COLUMN_HEADER" && header.value === "null" && (firstColumn == null || col < firstColumn)) {
282                     contents += '<th class="all_null"><div> </div></th>';
283                 } // If the cell is a column header and isn't null (column header of table)
284                 else if (header.type === "COLUMN_HEADER") {
285                     if (firstColumn == null) {
286                         firstColumn = col;
287                     }
288                     if (table[row].length == col+1)
289                         isLastColumn = true;
290                     else
291                         nextHeader = data[row][col+1];
292 
293 
294                     if (isLastColumn) {
295                         // Last column in a row....
296                         contents += '<th class="col" style="text-align: center;" colspan="' + colSpan + '" title="' + header.value + '"><div rel="' + row + ":" + col +'">' + header.value + '</div></th>';
297                     } else {
298                         // All the rest...
299                         var groupChange = (col > 1 && row > 1 && !isHeaderLowestLvl && col > firstColumn) ?
300                             data[row-1][col+1].value != data[row-1][col].value
301                             : false;
302                         if (header.value != nextHeader.value || isHeaderLowestLvl || groupChange) {
303                             if (header.value == "null") {
304                                 contents += '<th class="col_null" colspan="' + colSpan + '"><div> </div></th>';
305                             } else {
306                                 contents += '<th class="col" style="text-align: center;" colspan="' + (colSpan == 0 ? 1 : colSpan) + '" title="' + header.value + '"><div rel="' + row + ":" + col +'">' + header.value + '</div></th>';
307                             }
308                             colSpan = 1;
309                         } else {
310                             colSpan++;
311                         }
312                     }
313                 } // If the cell is a row header and is null (grouped row header)
314                 else if (header.type === "ROW_HEADER" && header.value === "null") {
315                     contents += '<th class="row_null"><div> </div></th>';
316                 } // If the cell is a row header and isn't null (last row header)
317                 else if (header.type === "ROW_HEADER") {
318                     if (lowestRowLvl == col)
319                         isHeaderLowestLvl = true;
320                     else
321                         nextHeader = data[row][col+1];
322 
323                     var previousRow = data[row - 1];
324 
325                     var same = !isHeaderLowestLvl && (col == 0 || previousRow[col-1].value == data[row][col-1].value) && header.value === previousRow[col].value;
326                     var value = (same ? "<div> </div>" : '<div rel="' + row + ":" + col +'">' + header.value + '</div>');
327                     var cssclass = (same ? "row_null" : "row");
328                     var colspan = 0;
329 
330                     if (!isHeaderLowestLvl && nextHeader.value === "null") {
331                         colspan = 1;
332                         var group = header.properties.dimension;
333                         var level = header.properties.level;
334                         var groupWidth = (group in rowGroups ? rowGroups[group].length - rowGroups[group].indexOf(level) : 1);
335                         for (var k = col + 1; colspan < groupWidth && data[row][k] !== "null"; k++) {
336                             colspan = k - col;
337                         }
338                         col = col + colspan -1;
339                     }
340                     contents += '<th class="' + cssclass + '" ' + (colspan > 0 ? ' colspan="' + colspan + '"' : "") + '>' + value + '</th>';
341                 }
342                 else if (header.type === "ROW_HEADER_HEADER") {
343                     contents += '<th class="row_header"><div>' + header.value + '</div></th>';
344                     isHeaderLowestLvl = true;
345                     processedRowHeader = true;
346                     lowestRowLvl = col;
347                     if (header.properties.hasOwnProperty("dimension")) {
348                         var group = header.properties.dimension;
349                         if (!(group in rowGroups)) {
350                             rowGroups[group] = [];
351                         }
352                         rowGroups[group].push(header.properties.level);
353                     }
354                 } // If the cell is a normal data cell
355                 else if (header.type === "DATA_CELL") {
356                     var color = "";
357                     var val = header.value;
358                     var arrow = "";
359                     if (header.properties.hasOwnProperty('style')) {
360                         color = " style='background-color: " + header.properties.style + "' ";
361                     }
362                     if (header.properties.hasOwnProperty('link')) {
363                         val = "<a target='__blank' href='" + header.properties.link + "'>" + val + "</a>";
364                     }
365                     if (header.properties.hasOwnProperty('arrow')) {
366                         arrow = "<img height='10' width='10' style='padding-left: 5px' src='/images/arrow-" + header.properties.arrow + ".gif' border='0'>";
367                     }
368 
369                     contents += '<td class="data" ' + color + '><div alt="' + header.properties.raw + '" rel="' + header.properties.position + '">' + val + arrow + '</div></td>';
370                 }
371             }
372             contents += "</tr>";
373             
374         }
375 
376         // Append the table
377         $(this.el).html(contents);
378         this.post_process();
379     },
380 
381     post_process: function() {
382         if (this.workspace.query.get('type') == 'QM' && Settings.MODE != "view") {
383             $(this.el).find('th.row, th.col').addClass('headerhighlight');
384         }
385     },
386     cancel: function(event) {
387         this.workspace.query.action.del("/result", {success: this.cancelled } );
388     },
389     
390     cancelled: function(args) {
391         $(this.el).html('<tr><td>No results</td></tr>');
392     },
393 
394     no_results: function(args) {
395         $(this.el).html('<tr><td>No results</td></tr>');
396     },
397     
398     error: function(args) {
399         $(this.el).html('<tr><td>' + args.data.error + '</td></tr>');
400     }
