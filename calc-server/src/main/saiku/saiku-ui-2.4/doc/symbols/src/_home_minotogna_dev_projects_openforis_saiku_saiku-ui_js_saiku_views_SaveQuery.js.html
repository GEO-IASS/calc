 21 /**
 22  * The save query dialog
 23  */
 24 var SaveQuery = Modal.extend({
 25     type: "save",
 26     closeText: "Save",
 27 
 28     events: {
 29         'click': 'select_root_folder', /* select root folder */
 30         'click .dialog_footer a:' : 'call',
 31         'submit form': 'save',
 32         'click .query': 'select_name',
 33         'click li.folder': 'toggle_folder'
 34     },
 35     
 36     buttons: [
 37         { text: "Save", method: "save" },
 38         { text: "Cancel", method: "close" }
 39     ],
 40 
 41     initialize: function(args) {
 42         // Append events
 43         var self = this;
 44         var name = "";
 45         if (args.query.name) {
 46             name = args.query.name.split('/')[args.query.name.split('/').length -1];
 47         }
 48         this.query = args.query;
 49         this.message = _.template("<form id='save_query_form'>" +
 50             "<label for='name'>To save a new query, " + 
 51             "please select a folder and type a name in the text box below:</label><br />" +
 52             "<div class='RepositoryObjects'></div>" +
 53             "<input type='text' name='name' value='<%= name %>' />" +
 54             "</form>")({ name: name });
 55 
 56         _.extend(this.options, {
 57             title: "Save query"
 58         });
 59 
 60         // Initialize repository
 61         this.repository = new Repository({}, { dialog: this });
 62 
 63         this.bind( 'open', function( ) {
 64             var height = ( $( "body" ).height() / 2 ) + ( $( "body" ).height() / 6 );
 65             if( height > 420 ) {
 66                 height = 420;
 67             }
 68             $(this.el).find('.RepositoryObjects').height( height );
 69             $(this.el).dialog( 'option', 'position', 'center' );
 70             $(this.el).parents('.ui-dialog').css({ width: "500px" });
 71             self.repository.fetch( );
 72         } );
 73 
 74         // Maintain `this`
 75         _.bindAll( this, "copy_to_repository", "close", "toggle_folder", "select_name", "populate" );
 76     },
 77 
 78     populate: function( repository ) {
 79         $( this.el ).find( '.RepositoryObjects' ).html(
 80             _.template( $( '#template-repository-objects' ).html( ) )( {
 81                 repoObjects: repository
 82             } ) 
 83         );
 84     },
 85 
 86     select_root_folder: function( event ) {
 87         var isNameInputField = $( event.target ).attr( 'name' ) === 'name';
 88         if( !isNameInputField ) {
 89             this.unselect_current_selected_folder( );
 90         }
 91     },
 92 
 93     toggle_folder: function( event ) {
 94         var $target = $( event.currentTarget );
 95         this.unselect_current_selected_folder( );
 96         $target.children('.folder_row').addClass( 'selected' );
 97         var $queries = $target.children( '.folder_content' );
 98         var isClosed = $target.children( '.folder_row' ).find('.sprite').hasClass( 'collapsed' );
 99         if( isClosed ) {
100             $target.children( '.folder_row' ).find('.sprite').removeClass( 'collapsed' );
101             $queries.removeClass( 'hide' );
102         } else {
103             $target.children( '.folder_row' ).find('.sprite').addClass( 'collapsed' );
104             $queries.addClass( 'hide' );
105         }
106         return false;
107     },
108 
109     select_name: function( event ) {
110         var $currentTarget = $( event.currentTarget );
111         this.unselect_current_selected_folder( );
112         $currentTarget.parent( ).parent( ).has( '.folder' ).children('.folder_row').addClass( 'selected' );
113         var name = $currentTarget.find( 'a' ).text();
114         $(this.el).find('input[name="name"]').val( name );
115         return false;
116     },
117 
118     unselect_current_selected_folder: function( ) {
119         $( this.el ).find( '.selected' ).removeClass( 'selected' );
120     },
121 
122     save: function(event) {
123         // Save the name for future reference
124         var foldername = ''; /* XXX == root, should it be something different than ''? */
125         var $folder = $(this.el).find( '.folder_row.selected a' ).first( );
126         if( $folder.length ) {
127             foldername = $folder.attr( 'href' ).replace( '#', '' );
128             foldername = (foldername != null && foldername.length > 0 ? foldername + "/" : "");
129         }
130         
131         var name = $(this.el).find('input[name="name"]').val();
132         this.query.set({ name: name, folder: foldername });
133         this.query.trigger('query:save');
134         $(this.el).find('form').html("Saving query...");
135         
136         // Fetch query XML and save to repository
137         this.query.action.get("/xml", {
138             success: this.copy_to_repository
139         });
140         
141         event.preventDefault();
142         return false;
143     },
144     
145     copy_to_repository: function(model, response) {
146         var folder = this.query.get('folder');
147         var file = this.query.get('name').indexOf(".saiku") == this.query.get('name').length - 6 ? this.query.get('name') : this.query.get('name') + ".saiku";
148         file = folder + file;
149         (new SavedQuery({
150             name: this.query.get('name'),
151             file: file,
152             content: response.xml
153         })).save({ success: this.close });
154     }
155 });
